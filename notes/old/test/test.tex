{\bf 
            Preliminaries
         }\end{flushleft}

    
    
    \begin{flushleft} 
 For $\A$ a $\sigma$-algebra, $E \in \A$, define $\partset{E, \A} \defeq \{ \{ E_k \in \A; k \in \nats \}; E = \cup_k E_k, E_i \cap E_j = \phi \;  \forall i \not = j \}$. Always $ \{ \phi, E \} \in \partset{E, \A}$, so $\partset{E, \A}$ is never empty, and also $\partset{\phi, \A} = \{ \phi \}$.\end{flushleft}
    \begin{flushleft} 
 
        For $\A$ a $\sigma$-algebra, define
        \begin{equation} 
 S^\pm(\A) \defeq \{ \sum_{k=1}^n c_k \, \cf{E_k}; \{c_k \in\reals; c_i \not = c_j \, \forall i \not = j\}, \{E_k \in \A; E_i \cap E_j = \phi \, \forall i \not = j \}, n \in \nats \}\end{equation}
        \begin{equation} 
 S^{+}(\A) \defeq \{ \sum_{k=1}^n c_k \, \cf{E_k}; \{c_k \in [0, \infty], c_i \not = c_j \, \forall i \not = j\}, \{E_k \in \A; E_i \cap E_j = \phi \, \forall i \not = j \}, n \in \nats \}\end{equation}
    \end{flushleft}
